\begin{sagesilent}

t=ZZ.random_element(999999)
set_random_seed(t)
numberOfQuestions = 60
largestPositiveInteger = 100
setOfPositiveIntegers = []
while len(setOfPositiveIntegers) < numberOfQuestions:
  generatedPositiveInteger = (1)^(ZZ.random_element(1,3))*(ZZ.random_element(2,largestPositiveInteger+1))
  if is_prime(generatedPositiveInteger) == False and (generatedPositiveInteger in setOfPositiveIntegers) == False :
    setOfPositiveIntegers.append(generatedPositiveInteger)

qoutput = ""
aoutput = ""

question = [latex(setOfPositiveIntegers[i]) for i in range(len(setOfPositiveIntegers))]
answer = [latex(factor(setOfPositiveIntegers[i])) for i in range(len(setOfPositiveIntegers))]

for i in range(len(setOfPositiveIntegers)):
  qoutput += r"\item $%s$" %(question[i])
  aoutput += r"\item \cRed{$%s$}" %(answer[i])

\end{sagesilent}
